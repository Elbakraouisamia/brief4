VALIDER UNE DATE

Enoncé

Ecrire un script qui permet de saisir les trois champs d’une date (jour, mois et année) et
d’indiquer si la date est correcte ou non

Démo




CODE SOURCE

function btn1() { let day = parseInt(document.getElementById('date').value); let month = parseInt(document.getElementById('month').value); let year = parseInt(document.getElementById('year').value); if (day <= 31 && month <= 12 && month !== 2 && month !== 4 && month !== 6 && month !== 9 && month !== 11 && year <= 2022 && year >= 1999){ document.getElementById('incorrecte').innerHTML = day + "/" + month + "/" + year + " est correcte" } else if(day <= 30 && month == 4 && month == 6 && month == 9 && month == 11 && year <= 2022 && year >= 2000){ document.getElementById('incorrecte').innerHTML = day + "/" + month + "/" + year + " est correcte !!" } else if(day < 29 && month == 02 && year <= 2022 && year >= 2000){ document.getElementById('incorrecte').innerHTML = day + "/" + month + "/" + year + " est correcte !!" } else{ document.getElementById('incorrecte').innerHTML = day + "/" + month + "/" + year + " est incorrecte !!" } }
